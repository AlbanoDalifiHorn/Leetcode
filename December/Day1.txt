class Solution {
    public int[] plusOne(int[] digits) {
        for(int i = digits.length - 1; i >= 0; i--) {
            if (digits[i] < 9) {
                digits[i]++;
                return digits;
            }
            digits[i] = 0;
        }
        // if array looks like that: [9,9,9] -> [1,0,0,0]
        int[] res = new int[digits.length + 1];
        res[0] = 1;
        return res;
    }
}
