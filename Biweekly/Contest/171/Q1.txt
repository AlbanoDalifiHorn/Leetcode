class Solution {
    public boolean completePrime(int num) {
        // check if both conditions are prime numbers
        return (prefix(num) && suffix(num));
    }

    // check if every prefix is a valid prime or not
    public static boolean prefix(int num) {
        String n = String.valueOf(num);
        String pre = "";
        for (int i = 0; i < n.length(); i++) {
            pre = pre + n.charAt(i);
            if (!isPrime(Integer.parseInt(pre))) return false;
        }
        return true;
    }

    // check if every suffix is a valid prime or nor
    public static boolean suffix(int num) {
        String n = String.valueOf(num);
        String last = "";
        for (int i = n.length() - 1; i >= 0; i--) {
            last = n.charAt(i) + last;
            if (!isPrime(Integer.parseInt(last))) return false;
        }
        return true;
    }

    //check if a number is a prime
    public static boolean isPrime(int num) {
        if (num <= 1) return false;
        if (num <= 3) return true;

        if (num % 2 == 0 || num % 3 == 0) return false;

        for (int i = 5; i * i <= num; i+= 6) {
            if (num % i == 0 || num % (i + 2) == 0) return false;
        }
        return true;
    }
}
